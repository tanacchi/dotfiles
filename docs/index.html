#!/bin/bash # This is tanacchi's auto-setup-script PYTHON_VERSION="3.9.0" set -eu # Update repositories and packages # Set INSTALL_COMMAND="brew" if the OS is MacOS UPDATE_COMMAND="${INSTALL_COMMAND:-apt -y}" sudo $UPDATE_COMMAND update sudo $UPDATE_COMMAND upgrade # Setup pyenv if [ ! -d "${HOME}/.pyenv" ]; then curl https://raw.githubusercontent.com/tanacchi/dotfiles/master/scripts/tools/pyenv.sh | sh else echo "pyenv is already setup." fi export PYENV_ROOT="$HOME/.pyenv" export PATH="$PYENV_ROOT/bin:$PATH" eval "$(pyenv init -)" pyenv install -f ${PYTHON_VERSION} pyenv global ${PYTHON_VERSION} curl https://raw.githubusercontent.com/tanacchi/dotfiles/master/scripts/setup/autosetup.py | python
